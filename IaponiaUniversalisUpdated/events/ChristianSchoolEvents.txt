#Invite a Permanent Scholar
country_event = {
	id = christian_school_events.24
	title = christian_school_events.24.t
	desc = christian_school_events.24.d
	picture = MYSTIC_IN_PRAYER_eventPicture
	
	is_triggered_only = yes

	immediate = {
		disallow_baseline_invite_scholar = yes
	}

	option = {
		name = christian_school_events.24.a
		trigger = {
			NOT = { has_country_flag = can_always_invite_dominican_school }
			NOT = { has_country_flag = privilege_can_always_invite_dominican_school }
			NOT = {
				religious_school = {
					group = christian_group
					school = dominican_school
				}
			}
		}
		allow_baseline_invite_scholar_privilege = {
			religious_school = dominican_school
		}
	}

	option = {
		name = christian_school_events.24.b
		trigger = {
			NOT = { has_country_flag = can_always_invite_franciscan_school }
			NOT = { has_country_flag = privilege_can_always_invite_franciscan_school }
			NOT = {
				religious_school = {
					group = christian_group
					school = franciscan_school
				}
			}
		}
		allow_baseline_invite_scholar_privilege = {
			religious_school = franciscan_school
		}
	}

	option = {
		name = christian_school_events.24.c
		trigger = {
			NOT = { has_country_flag = can_always_invite_jesuit_school }
			NOT = { has_country_flag = privilege_can_always_invite_jesuit_school }
			NOT = {
				religious_school = {
					group = christian_group
					school = jesuit_school
				}
			}
		}
		allow_baseline_invite_scholar_privilege = {
			religious_school = jesuit_school
		}
	}

	after = {
		clr_country_flag = swaps_scholar_flag
	}
}
