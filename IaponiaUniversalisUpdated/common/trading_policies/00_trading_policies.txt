# First policy below is the default if no other selected

maximize_profit =
{
	can_select = {
		FROM = {
			has_trader = ROOT
		}
	}

	trade_power = {
		duration = -1
		power_modifier = 0.05
		key = maximize_profits
	}

	countries_with_merchant_modifier = {
		production_efficiency = 0.02
	}

	node_province_modifier = {
		trade_goods_size_modifier = 0.05 #broken?
	}
	
	center_of_reformation = no
	button_gfx = GFX_Trading_Policy_Max_Profit
}

hostile_trading =
{
	can_select = {
		FROM = {
			has_trader = ROOT
		}
	}

	# Also applied versus countries with the node as their home node
	countries_with_merchant_modifier = {
		spy_offence = 0.33
		privateer_efficiency = 0.33
		improve_relation_modifier = -0.15
	}

	trade_power = {
		duration = -1
		power_modifier = -0.15
		key = hostile_trading_debuff_loc
	}
	
	button_gfx = GFX_Trading_Policy_Hostile_Trading
}

improve_inland_routes =
{
	can_select = {
		FROM = {
			has_trader = ROOT
			trade_share = {
				country = ROOT
				share = 50
			}
		}
	}

	can_maintain = {
		FROM = {
			has_trader = ROOT
			trade_share = {
				country = ROOT
				share = 40
			}
		}
	}
	
	node_province_modifier = {
		local_friendly_movement_speed = 0.15 #broken?
		local_hostile_movement_speed = 0.15 #broken?
		local_build_cost = -0.10  #broken?
		#global_ship_trade_power = -0.25 #broken?
	}

	trade_power = {
		duration = -1
		power_modifier = -0.10
		key = inland_routes_debuff_loc
	}
	button_gfx = GFX_Trading_Policy_Improve_Routes
}

establish_communities =
{
	can_select = {
		FROM = {
			has_trader = ROOT
		}
	}
	countries_with_merchant_modifier = {
		ae_impact = -0.1
		improve_relation_modifier = 0.1
		#global_autonomy = 0.05 #doesn't seem to work
	}
	
	trade_power = {
		duration = -1
		power_modifier = -0.10
		key = establish_communities_debuff_loc
	}

	node_province_modifier = {
		siege_ability = 0.10
	}

	button_gfx = GFX_Trading_Policy_Establish_Communities
}

propagate_religion =
{
	unique = yes # Only one country can select this in a certain node.

	can_select = {		
		religion_group = christian_group
		dominant_religion = ROOT
		FROM = {
			has_trader = ROOT
			is_node_in_trade_company_region = yes
			trade_share = {
				country = ROOT
				share = 40
			}
		}
	}

	can_maintain = {
		religion_group = christian_group
		dominant_religion = ROOT
		FROM = {
			has_trader = ROOT
			is_node_in_trade_company_region = yes
			trade_share = {
				country = ROOT
				share = 25
			}
		}
	}

	show_alert = yes
	center_of_reformation = yes
	button_gfx = GFX_Trading_Policy_Propagate_Religion
}
