estate_burghers_expand_highways_in_y = {
	can_select = {
		adm_tech = 7
		any_owned_province = {
			development = 10
			is_state_core = root
			NOT = {has_building = highways}
			NOT = {has_building = post_stations}
			any_neighbor_province = {
				development = 10
				OR = {
					has_building = highways
					has_building = post_stations
				}
			}
			#redundant but wouldn't be if roads took up building slots
			num_free_building_slots = 0
		}
	}
	selection_weight = {
		factor = 1000
		modifier = {
			factor = 665
			always = yes
		}
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				development = 10
				is_state_core = root
				NOT = {has_building = highways}
				NOT = {has_building = post_stations}
				any_neighbor_province = {
					development = 10
					OR = {
						has_building = highways
						has_building = post_stations
					}
				}
				#redundant but wouldn't be if roads took up building slots
				num_free_building_slots = 0
			}
			save_event_target_as = agenda_province
		}
		event_target:agenda_province = {
			random_neighbor_province = {
				limit = {
					development = 10
					OR = {
						has_building = highways
						has_building = post_stations
					}
				}
				save_event_target_as = envied_province	
			}
		}
		
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_requirements = {
		event_target:agenda_province = {
			OR = {
				has_building = highways
				has_building = post_stations
			}
		}
    }
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = 10
		}
		event_target:agenda_province = {
			add_devastation = -25
			add_local_autonomy = -25
			if = {
				limit = {
					NOT = {
						has_province_modifier = agenda_road_business_booming
					}
				}
				add_province_modifier = {
					name = agenda_road_business_booming
					duration = 1825
				}
			}
		}
    }
    failing_effect = {
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_burghers_displeased
					value = 5
				}
			}
		}
        add_estate_loyalty_modifier = {
            estate = estate_burghers
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}
